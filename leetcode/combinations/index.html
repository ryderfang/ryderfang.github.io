}

func subsets(_ nums: [Int]) -> [[Int]] {

    return ans
}

func subsetsWithDup(_ nums: [Int]) -> [[Int]] {
    var ans = Set<[Int]>()
    ans.insert([])
    for num in nums {
        for item in ans {
            var tmp = item
            tmp.append(num)
            tmp.sort()
            ans.insert(tmp)
        }
    }
    return Array(ans)
}

func subsetsWithDup2(_ nums: [Int]) -> [[Int]] {
    guard !nums.isEmpty else { return [] }

    let n = nums.count
    var ans: [[Int]] = [[]]
    let a = nums.sorted()
    func _backTracking(_ index: Int, _ result: inout [Int]) {
        guard index < n else {
            return
        }
        for i in index..<n {
            if (i > index && a[i] == a[i-1]) {
                continue
            }
            result.append(a[i])
            ans.append(result)
            _backTracking(i + 1, &result)
            result.removeLast()
        }
    }
    
    var tmp: [Int] = []
    _backTracking(0, &tmp)
    return ans
}
